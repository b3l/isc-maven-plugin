37  /**
38   * Deploys a collection of {@link Module}s to the Maven repository location indicated by the given {@link #repositoryUrl} property. 
39   * Functionally, pretty much just like the Deploy Plugin's deploy-file goal, except this one works on a collection.
40   * 
41   * @see http://maven.apache.org/plugins/maven-deploy-plugin/deploy-file-mojo.html
42   */
43  @Mojo(name="deploy", requiresProject=false)
44  public final class DeployMojo extends AbstractPackagerMojo {
45  	
46  	private static final Logger LOGGER = LoggerFactory.getLogger(DeployMojo.class);
47  	
48  	/**
49  	 * The identifier of a server entry from which Maven should read the authentication credentials
50  	 * to be used during deployment to {@link #repositoryUrl}. 
51  	 * 
52  	 * @see http://maven.apache.org/ref/3.1.1/maven-settings/settings.html#class_server
53  	 * @since 1.0.0
54  	 */
55  	@Parameter(property="repositoryId")
56  	private String repositoryId;
57  	
58  	/**
59  	 * The URL of the location to which the artifacts should be deployed.  e.g.,
60  	 * <p/>
61  	 * http://nexus.corp.int/nexus/content/repositories/thirdparty/
62  	 * 
63  	 * @since 1.0.0
64  	 */
65  	@Parameter(property="repositoryUrl", required=true)
66  	private String repositoryUrl;
67  	
68  	/**
69  	 * The repositoryType, as required by the Builder(String, String, String) constructor.
70  	 * 
71  	 * @see <a href="http://http://download.eclipse.org/aether/aether-core/0.9.0.M2/apidocs/org/eclipse/aether/repository/RemoteRepository.Builder.html#RemoteRepository.Builder(java.lang.String,%20java.lang.String,%20java.lang.String)">
72  	 * @since 1.0.0
73  	 */
74  	@Parameter(property="repositoryType", defaultValue="default")
75  	private String repositoryType;
76  	
77  	/**
78  	 * Deploy each of the provided {@link Module}s, along with their SubArtifacts (POMs, JavaDoc bundle, etc.), to the repository location
79  	 * indicated by {@link #repositoryUrl}.
80  	 */
81  	@Override 
82  	public void doExecute(Set<Module> artifacts) throws MojoExecutionException, MojoFailureException {
83  	
84  		for (Module artifact : artifacts) {
85  	     
86  			DeployRequest deployRequest = new DeployRequest();
87  			deployRequest.addArtifact(artifact);
88  	        
89  	        for (Artifact subArtifact : artifact.getAttachments()) {
90  	        	deployRequest.addArtifact(subArtifact);
91  	        }
92  
93  	        try {
94  	        	RemoteRepository repository = new RemoteRepository.Builder(repositoryId, repositoryType, repositoryUrl).build();
95  	        	deployRequest.setRepository(repository);
96  				repositorySystem.deploy(repositorySystemSession, deployRequest);
97  			} catch (DeploymentException e) {
98  				throw new MojoFailureException("Deployment failed: ", e);
99  			}	
100 
101 		}		
102 	}
103 }
