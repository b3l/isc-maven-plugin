134 	 * Object creation method meant to facilitate a 'fluent interface' style of configuration via a handful of private methods.
135 	 * Objects are created with a set of default property values and retruned for further configuration if so deired.  
136 	 * A reference to the return value is also available in the {@link #DISTRIBUTIONS} table under the given product/key combination.
137 	 * 
138 	 * @param product
139 	 * @param license
140 	 * @param links
141 	 * @return A Distribution object having default configuration values applied 
142 	 */
143 	private static Distribution create(Product product, License license, String... links) {
144 		
145 		Distribution distribution = new Distribution();
146 		distribution.include(links);
147 
148 		//smartclient downloads currently include smartgwt poms.  ignore when inapplicable 
149 		List<String> pomIncludes = new ArrayList<String>();
150 		if (product == SMARTGWT) {
151 			pomIncludes.add(POM_CLIENT);
152 			pomIncludes.add("**/smartgwt-" + license.name().toLowerCase() + ".pom");
153 		}
154 		//similarly lgpl includes server framework poms.  ignore
155 		if (license != LGPL) {
156 			pomIncludes.add(POM_SERVER);
157 		}
158 		pomIncludes.add(POM_SHARED);
159 
160 		/*
161 		 * Map the relevant sdk resources to their paths as they should be when extracted.
162 		 * Note that some resources are renamed to provide clarity, facilitate POM matching,
163 		 * and/or adhere to conventions.   
164 		 */
165 		distribution
166 			.contents("pom", Joiner.on(",").join(pomIncludes), null)
167 			.contents("doc/user", DOC_INCLUDES, DOC_EXCLUDES)
168 			.contents("assembly/selenium-resources", SELENIUM_INCLUDES, null)
169 			.contents("doc/api/client/#javadoc", SMARTGWT_CLIENT_JAVADOC, null)
170 			.contents("doc/api/server/#javadoc", product == SMARTGWT ? SMARTGWT_SERVER_JAVADOC : SMARTCLIENT_JAVADOC, null)
171 			.contents("lib", JAR_INCLUDES, product == SMARTCLIENT ? JAR_INCLUDES : JAR_EXCLUDES + ", " + JAR_CONFLICTS)
172 			.contents("lib/isc-jakarta-oro.jar", "**/isc-jakarta-oro*.jar", null)
173 			.contents("lib/smartgwt-analytics.jar", "**/analytics.jar", null)
174 			.contents("lib/smartgwt-messaging.jar", "**/messaging.jar", null)
175 			.contents("lib/isomorphic-messaging.jar", "**/isomorphic_realtime_messaging.jar", null);
176 		
177 		DISTRIBUTIONS.put(product, license, distribution);
178 		return distribution;
179 	}

182 	 * Returns a fully-prepared Distribution instance representing an Isomorphic build, suitable for invoking download or repackaging 
183 	 * operations against SDK bundles and their contents.
184 	 * 
185 	 * @param product The propduct for which a distribution exists and has been previously configured
186 	 * @param license The license under which the product is distributed and with which to execute some repackaing operation
187 	 * @param version The version number of the Isomorphic build, as described at smartclient.com
188 	 * @param date The date on which the build was made available
189 	 * @return The fully-prepared Distribution instance, suitable for downloading and/or manipulating SDK bundles and their contents
190 	 * @throws IllegalArgumentException when no instance is found for the given Product, License combination
191 	 */
192 	public static Distribution get(Product product, License license, String version, String date) {
193 		Distribution result = DISTRIBUTIONS.get(product, license);
194 		if (result == null) {
195 			throw new IllegalArgumentException("Unknown distribution for product " + product + " and license " + license + ".");
196 		}
197 		//determine the location of the distribution by replacing url tokens with provided values 
198 		String url = result.getRemoteIndex()
199 			.replaceAll("#product", product.toString())
200 			.replaceAll("#version", version)
201 			.replaceAll("#license", license.toString())
202 			.replaceAll("#date", date);
203 
204 		result.setRemoteIndex(url);
205 		return result;
206 	}

257 	public void unpack(File to) throws IOException {
258 		
259 		outer:
260 		for (File file : files) {
261 			String ext = FilenameUtils.getExtension(file.getName()).toUpperCase();
262 			if (! "ZIP".equals(ext)) {
263 				for(Map.Entry<String, AntPathMatcherFilter> filterEntry : content.entrySet()) {
264 					AntPathMatcherFilter filter = filterEntry.getValue();
265 					if (filter.accept(file.getName())) {
266 						File target = FileUtils.getFile(to, ArchiveUtils.rewritePath(file.getName(), filterEntry.getKey()));
267 						FileUtils.copyFile(file, target);
268 						LOGGER.debug("Copied file '{}' to file '{}'", file.getName(), target.getAbsolutePath());
269 						continue outer;
270 					}
271 				}
272 				FileUtils.copyFileToDirectory(file, new File(to, "lib"));
273 				continue outer;
274 			}
275 
276 			ZipFile zip = new ZipFile(file);
277 			Enumeration<? extends ZipEntry> entries = zip.entries();
278 			
279 			while (entries.hasMoreElements()) {
280 				ZipEntry entry = entries.nextElement();
281 				if (entry.isDirectory()) {
282 					continue;
283 				}
284 				for(Map.Entry<String, AntPathMatcherFilter> filterEntry : content.entrySet()) {
285 					AntPathMatcherFilter filter = filterEntry.getValue();
286 					if (filter.accept(entry.getName())) {
287 						File target = FileUtils.getFile(to, ArchiveUtils.rewritePath(entry.getName(), filterEntry.getKey()));
288 						FileUtils.copyInputStreamToFile(zip.getInputStream(entry), target);
289 						LOGGER.debug("Copied input stream to file '{}'", target.getAbsolutePath());
290 					}	
291 				}
292 			}
293 			zip.close();
294 		}
295 	
296 		File assembliesDir = new File(to, "assembly");
297 		//FileUtils.forceMkdir(assembliesDir);
298 	
299 		@SuppressWarnings("unchecked")
300 		Collection<File> assemblies = CollectionUtils.arrayToList(assembliesDir.listFiles(new FileFilter() {
301 			@Override
302 			public boolean accept(File arg0) {
303 				return arg0.isDirectory();
304 			}
305 		}));
306 		for (File assembly : assemblies) {
307 			String name = FilenameUtils.getBaseName(assembly.getName());
308 			LOGGER.debug("Copying resources for assembly '{}'", name);
309 			ArchiveUtils.zip(assembly, FileUtils.getFile(assembliesDir, "isc-" + name + ".zip"));
310 			FileUtils.deleteQuietly(assembly);
311 		}
312 
313 		LOGGER.debug("Repackaging Javadoc...");
314 		File docLib = new File(to, "doc/lib");
315 //		FileUtils.forceMkdir(docLib);
316 		
317 		//TODO at least make these directories available as member variables
318 		File client = FileUtils.getFile(to, "doc/api/client");
319 		if (client.exists()) {
320 			ArchiveUtils.jar(client, new File(docLib, "smartgwt-javadoc.jar"));	
321 		}
322 		
323 		File server = FileUtils.getFile(to, "doc/api/server");
324 		if (server.exists()) {
325 			ArchiveUtils.jar(server, new File(docLib, "isomorphic-javadoc.jar"));	
326 		}
327 	}
